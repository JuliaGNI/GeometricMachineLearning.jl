\documentclass[border = .2cm]{standalone}
 
\usepackage{pgfplots}

\pgfplotsset{compat = newest}

        },
    colormap={whiteblack}{color=(black) color=(white)},

    \begin{axis}[

        ymax = +3,

        ]

        ] {x^2 - y^2};
